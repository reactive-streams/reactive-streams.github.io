168   * in such it is different from {@link org.reactivestreams.tck.TestEnvironment#flop} which only records the error.

177    try {
178      fail(msg);
179    } catch (Throwable t) {
180      asyncErrors.add(t);

182    }
183    return null; // unreachable, the previous block will always exit by throwing

251  /**
252   * Looks for given {@code method} method in stack trace.
253   * Can be used to answer questions like "was this method called from onComplete?".
254   *
255   * @return the caller's StackTraceElement at which he the looked for method was found in the call stack, EMPTY otherwise
256   */
257  public Optional<StackTraceElement> findCallerMethodInStackTrace(String method) {
258    final Throwable thr = new Throwable(); // gets the stacktrace
259
260    for (StackTraceElement stackElement : thr.getStackTrace()) {
261      if (stackElement.getMethodName().equals(method)) {
262        return Optional.of(stackElement);
263      }
264    }
265    return Optional.empty();
266  }

480  public static class ManualSubscriberWithSubscriptionSupport<T> extends ManualSubscriber<T> {
481
482    public ManualSubscriberWithSubscriptionSupport(TestEnvironment env) {
483      super(env);
484    }
485
486    @Override
487    public void onNext(T element) {
488      env.debug(String.format("%s::onNext(%s)", this, element));
489      if (subscription.isCompleted()) {
490        super.onNext(element);
491      } else {
492        env.flop(String.format("Subscriber::onNext(%s) called before Subscriber::onSubscribe", element));
493      }
494    }
495
496    @Override
497    public void onComplete() {
498      env.debug(this + "::onComplete()");
499      if (subscription.isCompleted()) {
500        super.onComplete();
501      } else {
502        env.flop("Subscriber::onComplete() called before Subscriber::onSubscribe");
503      }
504    }
505
506    @Override
507    public void onSubscribe(Subscription s) {
508      env.debug(String.format("%s::onSubscribe(%s)", this, s));
509      if (!subscription.isCompleted()) {
510        subscription.complete(s);
511      } else {
512        env.flop("Subscriber::onSubscribe called on an already-subscribed Subscriber");
513      }
514    }
515
516    @Override
517    public void onError(Throwable cause) {
518      env.debug(String.format("%s::onError(%s)", this, cause));
519      if (subscription.isCompleted()) {
520        super.onError(cause);
521      } else {
522        env.flop(cause, String.format("Subscriber::onError(%s) called before Subscriber::onSubscribe", cause));
523      }
524    }
525  }

597  public static class ManualPublisher<T> implements Publisher<T> {
598    protected final TestEnvironment env;
599
600    protected long pendingDemand = 0L;
601    protected Promise<Subscriber<? super T>> subscriber;
602
603    protected final Receptacle<Long> requests;
604
605    protected final Latch cancelled;
606
607    public ManualPublisher(TestEnvironment env) {
608      this.env = env;
609      requests = new Receptacle<Long>(env);
610      cancelled = new Latch(env);
611      subscriber = new Promise<Subscriber<? super T>>(this.env);
612    }
613
614    @Override
615    public void subscribe(Subscriber<? super T> s) {
616      if (!subscriber.isCompleted()) {
617        subscriber.completeImmediatly(s);
618
619        Subscription subs = new Subscription() {
620          @Override
621          public void request(long elements) {
622            requests.add(elements);
623          }
624
625          @Override
626          public void cancel() {
627            cancelled.close();
628          }
629        };
630        s.onSubscribe(subs);
631
632      } else {
633        env.flop("TestPublisher doesn't support more than one Subscriber");
634      }
635    }
636
637    public void sendNext(T element) {
638      if (subscriber.isCompleted()) {
639        subscriber.value().onNext(element);
640      } else {
641        env.flop("Cannot sendNext before having a Subscriber");
642      }
643    }
644
645    public void sendCompletion() {
646      if (subscriber.isCompleted()) {
647        subscriber.value().onComplete();
648      } else {
649        env.flop("Cannot sendCompletion before having a Subscriber");
650      }
651    }
652
653    public void sendError(Throwable cause) {
654      if (subscriber.isCompleted()) {
655        subscriber.value().onError(cause);
656      } else {
657        env.flop("Cannot sendError before having a Subscriber");
658      }
659    }
660
661    public long expectRequest() throws InterruptedException {
662      return expectRequest(env.defaultTimeoutMillis());
663    }
664
665    public long expectRequest(long timeoutMillis) throws InterruptedException {
666      long requested = requests.next(timeoutMillis, "Did not receive expected `request` call");
667      if (requested <= 0) {
668        return env.<Long>flopAndFail(String.format("Requests cannot be zero or negative but received request(%s)", requested));
669      } else {
670        pendingDemand += requested;
671        return requested;
672      }
673    }
674
675    public void expectExactRequest(long expected) throws InterruptedException {
676      expectExactRequest(expected, env.defaultTimeoutMillis());
677    }
678
679    public void expectExactRequest(long expected, long timeoutMillis) throws InterruptedException {
680      long requested = expectRequest(timeoutMillis);
681      if (requested != expected) {
682        env.flop(String.format("Received `request(%d)` on upstream but expected `request(%d)`", requested, expected));
683      }
684      pendingDemand += requested;
685    }
686
687    public void expectNoRequest() throws InterruptedException {
688      expectNoRequest(env.defaultTimeoutMillis());
689    }
690
691    public void expectNoRequest(long timeoutMillis) throws InterruptedException {
692      requests.expectNone(timeoutMillis, "Received an unexpected call to: request: ");
693    }
694
695    public void expectCancelling() throws InterruptedException {
696      expectCancelling(env.defaultTimeoutMillis());
697    }
698
699    public void expectCancelling(long timeoutMillis) throws InterruptedException {
700      cancelled.expectClose(timeoutMillis, "Did not receive expected cancelling of upstream subscription");
701    }

704  /**
705   * Like a CountDownLatch, but resettable and with some convenience methods
706   */
707  public static class Latch {
708    private final TestEnvironment env;
709    volatile private CountDownLatch countDownLatch = new CountDownLatch(1);
710
711    public Latch(TestEnvironment env) {
712      this.env = env;
713    }
714
715    public void reOpen() {
716      countDownLatch = new CountDownLatch(1);
717    }
718
719    public boolean isClosed() {
720      return countDownLatch.getCount() == 0;
721    }
722
723    public void close() {
724      countDownLatch.countDown();
725    }
726
727    public void assertClosed(String openErrorMsg) {
728      if (!isClosed()) {
729        env.flop(new ExpectedClosedLatchException(openErrorMsg));
730      }
731    }
732
733    public void assertOpen(String closedErrorMsg) {
734      if (isClosed()) {
735        env.flop(new ExpectedOpenLatchException(closedErrorMsg));
736      }
737    }
738
739    public void expectClose(String notClosedErrorMsg) throws InterruptedException {
740      expectClose(env.defaultTimeoutMillis(), notClosedErrorMsg);
741    }
742
743    public void expectClose(long timeoutMillis, String notClosedErrorMsg) throws InterruptedException {
744      countDownLatch.await(timeoutMillis, TimeUnit.MILLISECONDS);
745      if (countDownLatch.getCount() > 0) {
746        env.flop(String.format("%s within %d ms", notClosedErrorMsg, timeoutMillis));
747      }
748    }
749
750    static final class ExpectedOpenLatchException extends RuntimeException {
751      public ExpectedOpenLatchException(String message) {
752        super(message);
753      }
754    }
755
756    static final class ExpectedClosedLatchException extends RuntimeException {
757      public ExpectedClosedLatchException(String message) {
758        super(message);
759      }
760    }

764  // simple promise for *one* value, which cannot be reset
765  public static class Promise<T> {
766    private final TestEnvironment env;
767
768    public static <T> Promise<T> completed(TestEnvironment env, T value) {
769      Promise<T> promise = new Promise<T>(env);
770      promise.completeImmediatly(value);
771      return promise;
772    }
773
774    public Promise(TestEnvironment env) {
775      this.env = env;
776    }
777
778    private ArrayBlockingQueue<T> abq = new ArrayBlockingQueue<T>(1);
779    private volatile T _value = null;
780
781    public T value() {
782      if (isCompleted()) {
783        return _value;
784      } else {
785        env.flop("Cannot access promise value before completion");
786        return null;
787      }
788    }
789
790    public boolean isCompleted() {
791      return _value != null;
792    }
793
794    /**
795     * Allows using expectCompletion to await for completion of the value and complete it _then_
796     */
797    public void complete(T value) {
798      abq.add(value);
799    }
800
801    /**
802     * Completes the promise right away, it is not possible to expectCompletion on a Promise completed this way
803     */
804    public void completeImmediatly(T value) {
805      complete(value); // complete!
806      _value = value;  // immediatly!
807    }
808
809    public void expectCompletion(long timeoutMillis, String errorMsg) throws InterruptedException {
810      if (!isCompleted()) {
811        T val = abq.poll(timeoutMillis, TimeUnit.MILLISECONDS);
812
813        if (val == null) {
814          env.flop(String.format("%s within %d ms", errorMsg, timeoutMillis));
815        } else {
816          _value = val;
817        }
818      }
819    }
820  }

822  // a "Promise" for multiple values, which also supports "end-of-stream reached"
823  public static class Receptacle<T> {
824    final int QUEUE_SIZE = 2 * TEST_BUFFER_SIZE;
825    private final TestEnvironment env;
826
827    private final ArrayBlockingQueue<Optional<T>> abq = new ArrayBlockingQueue<Optional<T>>(QUEUE_SIZE);
828
829    private final Latch completedLatch;
830
831    Receptacle(TestEnvironment env) {
832      this.env = env;
833      this.completedLatch = new Latch(env);
834    }
835
836    public void add(T value) {
837      completedLatch.assertOpen(String.format("Unexpected element %s received after stream completed", value));
838
839      abq.add(Optional.of(value));
840    }
841
842    public void complete() {
843      completedLatch.assertOpen("Unexpected additional complete signal received!");
844      completedLatch.close();
845
846      abq.add(Optional.<T>empty());
847    }
848
849    public T next(long timeoutMillis, String errorMsg) throws InterruptedException {
850      Optional<T> value = abq.poll(timeoutMillis, TimeUnit.MILLISECONDS);
851
852      if (value == null) {
853        return env.flopAndFail(String.format("%s within %d ms", errorMsg, timeoutMillis));
854      } else if (value.isDefined()) {
855        return value.get();
856      } else {
857        return env.flopAndFail("Expected element but got end-of-stream");
858      }
859    }
860
861    public Optional<T> nextOrEndOfStream(long timeoutMillis, String errorMsg) throws InterruptedException {
862      Optional<T> value = abq.poll(timeoutMillis, TimeUnit.MILLISECONDS);
863
864      if (value == null) {
865        env.flop(String.format("%s within %d ms", errorMsg, timeoutMillis));
866        return Optional.empty();
867      }
868
869      return value;
870    }
871
872    /**
873     * @param timeoutMillis total timeout time for awaiting all {@code elements} number of elements
874     */
875    public List<T> nextN(long elements, long timeoutMillis, String errorMsg) throws InterruptedException {
876      List<T> result = new LinkedList<T>();
877      long remaining = elements;
878      long deadline = System.currentTimeMillis() + timeoutMillis;
879      while (remaining > 0) {
880        long remainingMillis = deadline - System.currentTimeMillis();
881
882        result.add(next(remainingMillis, errorMsg));
883        remaining--;
884      }
885
886      return result;
887    }
888
889
890    public void expectCompletion(long timeoutMillis, String errorMsg) throws InterruptedException {
891      Optional<T> value = abq.poll(timeoutMillis, TimeUnit.MILLISECONDS);
892
893      if (value == null) {
894        env.flop(String.format("%s within %d ms", errorMsg, timeoutMillis));
895      } else if (value.isDefined()) {
896        env.flop(String.format("Expected end-of-stream but got element [%s]", value.get()));
897      } // else, ok
898    }
899
900    @SuppressWarnings("unchecked")
901    public <E extends Throwable> E expectError(Class<E> clazz, long timeoutMillis, String errorMsg) throws Exception {
902      Thread.sleep(timeoutMillis);
903
904      if (env.asyncErrors.isEmpty()) {
905        return env.flopAndFail(String.format("%s within %d ms", errorMsg, timeoutMillis));
906      } else {
907        // ok, there was an expected error
908        Throwable thrown = env.asyncErrors.remove(0);
909
910        if (clazz.isInstance(thrown)) {
911          return (E) thrown;
912        } else {
913
914          return env.flopAndFail(String.format("%s within %d ms; Got %s but expected %s",
915                                               errorMsg, timeoutMillis, thrown.getClass().getCanonicalName(), clazz.getCanonicalName()));
916        }
917      }
918    }
919
920    public void expectNone(long withinMillis, String errorMsgPrefix) throws InterruptedException {
921      Thread.sleep(withinMillis);
922      Optional<T> value = abq.poll();
923
924      if (value == null) {
925        // ok
926      } else if (value.isDefined()) {
927        env.flop(String.format("%s [%s]", errorMsgPrefix, value.get()));
928      } else {
929        env.flop("Expected no element but got end-of-stream");
930      }
931    }
